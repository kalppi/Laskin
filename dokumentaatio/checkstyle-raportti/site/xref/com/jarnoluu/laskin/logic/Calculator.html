13  /**
14   * Luokka, joka hoitaa laskujen laskemisen.
15   * @author Jarno Luukkonen <luukkonen.jarno@gmail.com>
16   */
17  public class Calculator {
18      /**
19       * Pitää sisällään operaattorien funktiot.
20       */
21      private final Map<String, String> operators = new HashMap();
22      
23      private final ScriptManager scriptManager;
24      
25      /**
26       * Edellisen laskun tulos.
27       */
28      private double lastValue = 0;
29      
30      public Calculator() throws LaskinScriptException {
31          this.scriptManager = new ScriptManager("JavaScript", "js/");
32          
33          this.scriptManager.loadScript("operators.js");
34          this.scriptManager.loadScript("functions.js");
35          
36          this.operators.put("+", "_plus");
37          this.operators.put("-", "_minus");
38          this.operators.put("*", "_mult");
39          this.operators.put("/", "_div");
40          this.operators.put("^", "_pow");
41          this.operators.put("%", "_mod");
42      }
43      
44      public double getLastValue() {
45          return this.lastValue;
46      }
47      
48      private double getSpecialValue(String special) {
49          switch (special) {
50              case "pi":
51                  return Math.PI;
52              case "$":
53                  return this.lastValue;
54              default:
55                  return 0;
56          }
57      }
58      
59      /**
60       * Laskee lausekkeen tuloksen.
61       * @param input laskulauseke merkkijonona
62       * @return tulos
63       * @throws LaskinInvalidArgumentException
64       * @throws LaskinParseException
65       * @throws LaskinCalculationException 
66       */
67      public double calculate(String input) throws LaskinInvalidArgumentException, LaskinParseException, LaskinCalculationException {
68          if (input == null || input.length() == 0) {
69              throw new LaskinInvalidArgumentException("Empty input");
70          }
71          
72          LinkedList<Token> tokens = Parser.tokenize(input);
73          
74          return this.calculate(tokens);
75      }
76      
77      /**
78       * Laskee lausekkeen tuloksen.
79       * @param tokens laskulauseke listana tokeneja
80       * @return tulos
81       * @throws LaskinInvalidArgumentException
82       * @throws LaskinParseException
83       * @throws LaskinCalculationException 
84       */
85      public double calculate(LinkedList<Token> tokens) throws LaskinInvalidArgumentException, LaskinParseException, LaskinCalculationException {
86          tokens = InfixToPostfix.transform(tokens);
87          
88          LinkedList<Double> stack = new LinkedList();
89          
90          while (tokens.size() > 0) {
91              Token t = tokens.remove(0);
92              String data = t.getData();
93              
94              switch (t.getType()) {
95                  case NUMBER:
96                      stack.add(Double.parseDouble(data));
97                      break;
98                  case NUMBER_HEX:
99                      stack.add(Util.hexToDouble(data));
100                     break;
101                 case NUMBER_BIN:
102                     stack.add(Util.binToDouble(data));
103                     break;
104                 case NUMBER_OCT:
105                     stack.add(Util.octToDouble(data));
106                     break;
107                 case SPECIAL:
108                     stack.add(this.getSpecialValue(data));
109                     break;
110                 case OPER:
111                     String func = this.operators.get(data);
112                     
113                     if (func == null) {
114                         throw new LaskinCalculationException("Unknown operator (" + data + ")");
115                     }
116                     
117                     stack.add(this.scriptManager.invokeFunction(func, stack));
118                     
119                     break;
120                 case FUNC:
121                     if (!this.scriptManager.functionExists(t.getData())) {
122                         throw new LaskinCalculationException("Unknown function (" + data + ")");
123                     }
124                     
125                     stack.add(this.scriptManager.invokeFunction(data, stack));
126                     
127                     break;
128             }
129         }
130         
131         if (stack.size() > 1) {
132             throw new LaskinCalculationException("Error");
133         } else if (stack.isEmpty()) {
134             return 0;
135         }
136         
137         double val = stack.get(0);
138         this.lastValue = val;
139         
140         return val;
141     }
142 }
