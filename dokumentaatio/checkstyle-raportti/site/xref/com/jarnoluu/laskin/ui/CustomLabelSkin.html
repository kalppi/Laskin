27  public class CustomLabelSkin implements Skin<Label> {
28      private final Label textField;
29      private final StackPane rootNode = new StackPane();
30      private final String fontName;
31      private final double fontSize;
32      
33      private static final int PADDING = 4;
34      
35      private final Background bg;
36      
37      private final Background bgError = new Background(
38          new BackgroundFill(Color.rgb(255, 230, 230), CornerRadii.EMPTY, Insets.EMPTY)
39      );
40      
41      public CustomLabelSkin(final Label textField, Color background) {
42          this.textField = textField;
43          
44          this.rootNode.alignmentProperty().setValue(Pos.CENTER_RIGHT);
45          this.rootNode.setPadding(new Insets(5, 5, 5, 5));
46          
47          Font font = this.textField.fontProperty().get();
48          this.fontName = font.getName();
49          this.fontSize = font.getSize();
50         
51          this.bg = new Background(
52              new BackgroundFill(background, CornerRadii.EMPTY, Insets.EMPTY)
53          );
54      }
55      
56      @Override
57      public Node getNode() {
58          this.draw(this.fontSize);
59          return this.rootNode;
60      }
61      
62      @Override
63      public void dispose() {
64      }
65  
66      @Override
67      public Label getSkinnable() {
68          return this.textField;
69      }
70      
71      private Node generateTextNode(String text, double fontSize, Color color) {
72          Text node = new Text(text);
73          node.setFont(new Font(this.fontName, fontSize));
74          node.setFill(color);
75          
76          return node;
77      }
78      
79      private Color getPartColor(Token.Type type) {
80          switch (type) {
81              case NUMBER:
82              case NUMBER_HEX:
83              case NUMBER_BIN:
84              case NUMBER_OCT:
85                  return Color.rgb(70, 70, 70);
86              case OPER:
87                  return Color.GREEN;
88              case FUNC:
89                  return Color.BLUE;
90              case BRACKET_START:
91              case BRACKET_END:
92                  return Color.BLUE;
93              case COMMA:
94                  return Color.BLUE;
95              case SPECIAL:
96                  return Color.MAGENTA;
97              case UNKNOWN:
98                  return Color.RED;
99              default:
100                 return Color.BLACK;
101         }
102     }
103     
104     public void draw(double fontSize) {
105         double w = this.textField.getWidth();
106         
107         if (this.textField.getPseudoClassStates().contains(PseudoClass.getPseudoClass("error"))) {
108             this.rootNode.setBackground(this.bgError);
109         } else {
110             this.rootNode.setBackground(this.bg);
111         }
112         
113         String te = this.textField.getText();
114         
115         Group textGroup = new Group();
116         
117         if (te.length() > 0) {
118             double x = w;
119             double y = 0;
120 
121             LinkedList<Token> parts;
122             
123             try {
124                 parts = Parser.tokenize(te, false, true);
125             } catch (Exception e) {
126                 return;
127             }
128             
129             for (Token t : parts) {
130                 if (t.getType() == Token.Type.UNKNOWN) {
131                     this.rootNode.setBackground(this.bgError);
132                     break;
133                 }
134             }
135 
136             double totalTextWidth = 0;
137 
138             while (parts.size() > 0) {
139                 Token part = parts.removeLast();
140                 Token.Type type = part.getType();
141                 
142                 if (type == Token.Type.NUMBER) {
143                     String data = part.getData();
144                     
145                     LinkedList<String> numberParts = Util.splitNumber(data, 3);
146                     
147                     Group numberGroup = new Group();
148                     int xx = 0;
149                     int numberTotalWidth = 0;
150 
151                     for (String numberPart : numberParts) {
152                         Node node = this.generateTextNode(numberPart, fontSize, this.getPartColor(Token.Type.NUMBER));
153 
154                         Bounds b = node.getLayoutBounds();
155 
156                         node.setLayoutX(xx);
157 
158                         xx += b.getWidth() + CustomLabelSkin.PADDING;
159                         numberTotalWidth += b.getWidth() + CustomLabelSkin.PADDING;
160 
161                         numberGroup.getChildren().add(node);
162                     }
163 
164                     totalTextWidth += numberTotalWidth;
165                     x -= numberTotalWidth;
166 
167                     numberGroup.setLayoutX(x);
168                     numberGroup.setLayoutY(y);
169 
170                     textGroup.getChildren().add(numberGroup);
171                 } else {
172                     Node node = this.generateTextNode(part.toString(), fontSize, this.getPartColor(type));
173 
174                     Bounds b = node.getLayoutBounds();
175                     x -= b.getWidth() + CustomLabelSkin.PADDING;
176                     totalTextWidth += b.getWidth() + CustomLabelSkin.PADDING;
177 
178                     node.setLayoutX(x);
179                     node.setLayoutY(y);
180 
181                     textGroup.getChildren().add(node);
182                 }
183                 
184                 if (totalTextWidth >= w - 5) {
185                     if (fontSize > 5) {
186                         this.draw(fontSize - 0.5);
187                     }
188                     return;
189                 }
190             }
191         }
192         
193         this.rootNode.getChildren().setAll(textGroup);
194     }
