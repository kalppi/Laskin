29  public class FancyLabelSkin implements Skin<FancyLabel> {
30      private static final int PADDING = 4;
31      
32      private final FancyLabel label;
33      private final StackPane rootNode = new StackPane();
34      private final Font font;
35      private final Background bg;
36      
37      private final Background bgError = new Background(
38          new BackgroundFill(Color.rgb(255, 230, 230), CornerRadii.EMPTY, Insets.EMPTY)
39      );
40      
41      public FancyLabelSkin(final FancyLabel label) {
42          this.label = label;
43          
44          this.rootNode.setAlignment(Pos.CENTER_RIGHT);
45          this.rootNode.setPadding(new Insets(5, 5, 5, 5));
46          this.font = this.label.getFont();
47          this.bg = this.label.getBackground();
48      }
49      
50      @Override
51      public Node getNode() {
52          this.render(this.font.getSize());
53          return this.rootNode;
54      }
55      
56      @Override
57      public void dispose() {
58      }
59  
60      @Override
61      public FancyLabel getSkinnable() {
62          return this.label;
63      }
64      
65      private Node generateTextNode(String text, double fontSize, Color color) {
66          Text node = new Text(text);
67          node.setFont(new Font(this.font.getName(), fontSize));
68          node.setFill(color);
69          
70          return node;
71      }
72      
73      private Color getPartColor(Token.Type type) {
74          switch (type) {
75              case NUMBER:
76              case NUMBER_HEX:
77              case NUMBER_BIN:
78              case NUMBER_OCT:
79                  return Color.rgb(70, 70, 70);
80              case OPER:
81                  return Color.GREEN;
82              case FUNC:
83                  return Color.BLUE;
84              case BRACKET_START:
85              case BRACKET_END:
86                  return Color.BLUE;
87              case COMMA:
88                  return Color.BLUE;
89              case SPECIAL:
90                  return Color.MAGENTA;
91              case UNKNOWN:
92                  return Color.RED;
93              default:
94                  return Color.BLACK;
95          }
96      }
97      
98      public void render(double fontSize) {
99          double w = this.rootNode.getWidth() - this.rootNode.getPadding().getLeft() - this.rootNode.getPadding().getRight();
100         
101         if (this.label.getPseudoClassStates().contains(PseudoClass.getPseudoClass("error"))) {
102             this.rootNode.setBackground(this.bgError);
103         } else {
104             this.rootNode.setBackground(this.bg);
105         }
106         
107         ObservableList<Node> children = this.rootNode.getChildren();
108         children.clear();
109         
110         CalculationString calc = this.label.getCalculation();
111         
112         if (calc == null) {
113             return;
114         }
115         
116         LinkedList<Token> parts = calc.getTokens();
117         Group textGroup = new Group();
118         textGroup.setTranslateY(2);
119         Rectangle selected = null;
120         
121         if (parts.size() > 0) {
122             double x = w;
123             
124             for (Token t : parts) {
125                 if (t.getType() == Token.Type.UNKNOWN) {
126                     this.rootNode.setBackground(this.bgError);
127                     break;
128                 }
129             }
130 
131             double totalTextWidth = 0;
132             ListIterator<Token> it = parts.listIterator(parts.size());
133             int selectedIndex = parts.size() - calc.getSelected() - 1;
134             
135             while (it.hasPrevious()) {
136                 Token part = it.previous();
137                 Token.Type type = part.getType();
138                 
139                 if (type == Token.Type.NUMBER) {
140                     String data = part.getData();
141                     
142                     LinkedList<String> numberParts = Util.splitNumber(data, 3);
143                     
144                     Group numberGroup = new Group();
145                     int xx = 0;
146 
147                     for (String numberPart : numberParts) {
148                         Node node = this.generateTextNode(numberPart, fontSize, this.getPartColor(Token.Type.NUMBER));
149                         node.setLayoutX(xx);
150                         
151                         Bounds b = node.getLayoutBounds();
152                         xx += b.getWidth() + FancyLabelSkin.PADDING;
153                         
154                         numberGroup.getChildren().add(node);
155                     }
156                     
157                     Bounds textBounds = numberGroup.getLayoutBounds();
158                     totalTextWidth += textBounds.getWidth() - FancyLabelSkin.PADDING;
159                     x -= textBounds.getWidth() + FancyLabelSkin.PADDING;
160 
161                     numberGroup.setLayoutX(x);
162 
163                     textGroup.getChildren().add(numberGroup);
164                     
165                     if (it.nextIndex() == selectedIndex) {
166                         selected = new Rectangle(x, this.label.getHeight() / 2 + textBounds.getMinY() + 3, textBounds.getWidth() + 3, textBounds.getHeight());
167                     }
168                 } else {
169                     Node node = this.generateTextNode(part.toPrettyString(), fontSize, this.getPartColor(type));
170                     
171                     Bounds textBounds = node.getLayoutBounds();
172                     x -= textBounds.getWidth() + FancyLabelSkin.PADDING;
173                     totalTextWidth += textBounds.getWidth() + FancyLabelSkin.PADDING + 8;
174 
175                     node.setLayoutX(x);
176                     textGroup.getChildren().add(node);
177                     
178                     if (it.nextIndex() == selectedIndex) {
179                         selected = new Rectangle(x, this.label.getHeight() / 2 + textBounds.getMinY() + 3, textBounds.getWidth() + 3, textBounds.getHeight());
180                     }
181                 }
182                 
183                 if (totalTextWidth >= w) {
184                     if (fontSize > 5) {
185                         this.render(fontSize - 0.5);
186                     }
187                     return;
188                 }
189             }
190         }
191         
192         if (selected != null) {
193             selected.setFill(Color.rgb(250, 195, 170));
194             selected.setTranslateX(selected.getX() - w + selected.getWidth() + 3);
195             children.add(selected);
196         }
197         
198         children.add(textGroup);
199     }
